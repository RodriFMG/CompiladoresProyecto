program FunctionCallsExample;

function FunctionA(x: Integer): Integer;
begin
  x := x + 2;
  result := FunctionB(x);
  FunctionA := result;
end;

function FunctionB(y: Integer): Integer;
begin
  y := y * 2;
  result := FunctionC(y);
  FunctionB := result;
end;

function FunctionC(z: Integer): Integer;
begin
  z := z - 3;
  FunctionC := z;
end;

var
  result: Integer;

begin
  result := FunctionA(5);
  writeln(result);
end.
